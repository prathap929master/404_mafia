Question 21: . Write a program to electricity bill using switch statement. (Hint: enter present and previous reading and calculate the monthly bill amount based on the following criteria ) 

import java.util.Scanner;

public class ElectricityBill21 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter previous reading: ");
        int prevReading = sc.nextInt();
        System.out.print("Enter present reading: ");
        int presReading = sc.nextInt();

        if (presReading < prevReading || prevReading < 0 || presReading < 0) {
            System.out.println("Invalid readings!");
            return;
        }

        int units = presReading - prevReading;
        double rate = 0.0;

        int slab;
        if (units <= 50) slab = 1;
        else if (units <= 100) slab = 2;
        else if (units <= 200) slab = 3;
        else if (units <= 250) slab = 4;
        else if (units <= 400) slab = 5;
        else if (units <= 500) slab = 6;
        else slab = 7;

        switch (slab) {
            case 1:
                rate = 3.50;
                break;
            case 2:
                rate = 3.95;
                break;
            case 3:
                rate = 6.80;
                break;
            case 4:
                rate = 8.00;
                break;
            case 5:
                rate = 7.35;
                break;
            case 6:
                rate = 7.60;
                break;
            case 7:
                rate = 8.50;
                break;
            default:
                System.out.println("Invalid slab!");
                return;
        }

        double billAmount = units * rate;

        System.out.println("\nElectricity Bill Details");
        System.out.println("--------------------------");
        System.out.println("Previous Reading : " + prevReading);
        System.out.println("Present Reading  : " + presReading);
        System.out.println("Units Consumed   : " + units);
        System.out.println("Rate per Unit    : â‚¹" + rate);
        System.out.println("Total Bill       : â‚¹" + billAmount);
    }

Question 22: Write a program to implement simple calculator using switch statements. Using for loop 

import java.util.Scanner;

public class SimpleCalculator22 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("How many calculations do you want to perform? ");
        int times = sc.nextInt();

        double result;

        for (int count = 1; count <= times; count++) {
            System.out.println("\n--- Calculation NO: " + count + " ---");

            System.out.print("Enter first number: ");
            double num1 = sc.nextDouble();
            System.out.print("Enter second number: ");
            double num2 = sc.nextDouble();

            System.out.println("Enter the operator you want to perform: (+, -, *, /)");
            char operator = sc.next().charAt(0);

            switch (operator) {
                case '+':
                    result = num1 + num2;
                    break;
                case '-':
                    result = num1 - num2;
                    break;
                case '*':
                    result = num1 * num2;
                    break;
                case '/':
                    if (num2 != 0) {
                        result = num1 / num2;
                    } else {
                        System.out.println("Error: Division by zero!");
                        continue; 
                    }
                    break;
                default:
                    System.out.println("Invalid choice!");
                    continue; 
            }

            System.out.println("Result = " + result);
        }

        System.out.println("\nCalculator session ended.");
        sc.close();
    }
}

Question 25: Write a program to print even numbers between two ranges. Using do... while() loop 

import java.util.Scanner;

public class EvenNumbersInRange25 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter start number: ");
        int start = sc.nextInt();

        System.out.print("Enter end number: ");
        int end = sc.nextInt();

        if (start % 2 != 0) {
            start++;
        }

        System.out.println("Even numbers between the range:");

        int num = start;
        do {
            System.out.print(num + " ");
            num += 2;
        } while (num <= end);

    }
}

Question 27: Write a program to find average of even numbers from a list of numbers 

import java.util.Scanner;

public class AverageOfEvenNumbers27 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter how many numbers you want to enter: ");
        int n = sc.nextInt();

        int sum = 0, countEven = 0;

        for (int i = 1; i <= n; i++) {
            System.out.print("Enter number " + i + ": ");
            int num = sc.nextInt();

            if (num % 2 == 0) {
                sum += num;    
                countEven++;  
            }
        }

        if (countEven > 0) {
            double average = (double) sum / countEven;
            System.out.println("Average of even numbers = " + average);
        } else {
            System.out.println("No even numbers entered.");
        }
    }
}

Question 29: Write a program to find the sum of even digits of a number. Using while loop 

import java.util.Scanner;

public class SumEvenDigits29 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter a number: ");
        int num = sc.nextInt();

        int sum = 0;

        while (num > 0) {
            int digit = num % 10; 
            if (digit % 2 == 0) { 
                sum += digit;
            }
            num /= 10; 
        }

        System.out.println("Sum of even digits: " + sum);
    }
}

        Scanner sc = new Scanner(System.in);

        // Input number
        System.out.print("Enter a number: ");
        int num = sc.nextInt();

        long factorial = 1; // use long to handle large results
        int i = 1;

        // While loop to calculate factorial
        while (i <= num) {

Question 31: Wite a program to find the prime numbers up to a limit. Using while loop. (hint: A prime number is a whole number greater than 1 whose only factors are 1 and itself example 2,3,5,7, 11 etc.). 

import java.util.Scanner;

public class PrimeNumbersWhileLoop31 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input limit
        System.out.print("Enter the limit: ");
        int limit = sc.nextInt();

        int num = 2; 

        while (num <= limit) {
            boolean isPrime = true;
            int i = 2;

            while (i <= num / 2) {
                if (num % i == 0) {
                    isPrime = false;
                    break; 
                }
                i++;
            }

            if (isPrime) {
                System.out.print(num + " ");
            }

            num++;
        }
    }
}
Question 32: Check whether the given number is palindrome or not. (hint: A palindrome number is a number that is same after reverse. For example 121, 34543 etc) 

import java.util.Scanner;

public class PalindromeNumber32 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter a number: ");
        int num = sc.nextInt();

        int originalNum = num;  
        int reversedNum = 0;

        while (num != 0) {
            int digit = num % 10;   
            reversedNum = reversedNum * 10 + digit; 
            num /= 10; 
        }

        if (originalNum == reversedNum) {
            System.out.println(originalNum + " is a Palindrome number.");
        } else {
            System.out.println(originalNum + " is NOT a Palindrome number.");
        }
    }
}
